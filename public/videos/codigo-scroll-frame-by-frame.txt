"use client";

import { useEffect, useRef } from "react";
import Lenis from "lenis";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

const TOTAL_FRAMES = 240;

function getFramePath(index: number) {
  const frameNumber = String(index + 1).padStart(5, "0");
  return `/frames/inital/frame-${frameNumber}.jpg`;
}

export default function Home() {
  const canvasRef = useRef<HTMLCanvasElement | null>(null);
  const containerRef = useRef<HTMLElement | null>(null);
  const textRef = useRef<HTMLDivElement | null>(null);

  useEffect(() => {
    const lenis = new Lenis({
      smoothWheel: true,
      lerp: 0.08,
    });

    function raf(time: number) {
      lenis.raf(time);
      ScrollTrigger.update();
      requestAnimationFrame(raf);
    }

    requestAnimationFrame(raf);

    return () => {
      lenis.destroy();
    };
  }, []);

  useEffect(() => {
    const canvas = canvasRef.current;
    const container = containerRef.current;

    if (!canvas || !container) return;

    const context = canvas.getContext("2d");

    if (!context) return;

    const frame = {
      index: 0,
    };

    const images: HTMLImageElement[] = [];
    let imagesLoaded = 0;

    const setCanvasSize = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    };

    const render = () => {
      const image = images[Math.round(frame.index)];

      if (!image) return;

      context.clearRect(0, 0, canvas.width, canvas.height);

      const canvasRatio = canvas.width / canvas.height;
      const imageRatio = image.width / image.height;

      let drawWidth = canvas.width;
      let drawHeight = canvas.height;
      let offsetX = 0;
      let offsetY = 0;

      if (imageRatio > canvasRatio) {
        drawHeight = canvas.height;
        drawWidth = image.width * (canvas.height / image.height);
        offsetX = (canvas.width - drawWidth) / 2;
      } else {
        drawWidth = canvas.width;
        drawHeight = image.height * (canvas.width / image.width);
        offsetY = (canvas.height - drawHeight) / 2;
      }

      context.drawImage(image, offsetX, offsetY, drawWidth, drawHeight);
    };

    setCanvasSize();

    for (let i = 0; i < TOTAL_FRAMES; i++) {
      const image = new Image();
      image.src = getFramePath(i);

      image.onload = () => {
        imagesLoaded++;

        if (imagesLoaded === 1) {
          render();
        }
      };

      images.push(image);
    }

    const ctx = gsap.context(() => {
      gsap.to(frame, {
        index: TOTAL_FRAMES - 1,
        ease: "none",
        snap: "index",
        scrollTrigger: {
          trigger: container,
          start: "top top",
          end: "+=5000",
          scrub: 1,
          pin: true,
        },
        onUpdate: render,
      });

      gsap.fromTo(
        ".intro-title",
        {
          opacity: 0,
          y: 80,
          filter: "blur(18px)",
        },
        {
          opacity: 1,
          y: 0,
          filter: "blur(0px)",
          duration: 1.2,
          ease: "power4.out",
        }
      );

      gsap.fromTo(
        ".intro-text",
        {
          opacity: 0,
          y: 40,
          filter: "blur(12px)",
        },
        {
          opacity: 1,
          y: 0,
          filter: "blur(0px)",
          duration: 1,
          delay: 0.25,
          ease: "power4.out",
        }
      );

      gsap.to(textRef.current, {
        opacity: 0,
        y: -80,
        filter: "blur(16px)",
        scrollTrigger: {
          trigger: container,
          start: "top top",
          end: "+=1200",
          scrub: true,
        },
      });

      gsap.fromTo(
        ".final-content",
        {
          opacity: 0,
          y: 80,
          filter: "blur(20px)",
        },
        {
          opacity: 1,
          y: 0,
          filter: "blur(0px)",
          scrollTrigger: {
            trigger: ".final-section",
            start: "top 70%",
            end: "top 25%",
            scrub: true,
          },
        }
      );
    });

    const handleResize = () => {
      setCanvasSize();
      render();
    };

    window.addEventListener("resize", handleResize);

    return () => {
      ctx.revert();
      window.removeEventListener("resize", handleResize);
      ScrollTrigger.killAll();
    };
  }, []);

  const scrollToExperience = () => {
    containerRef.current?.scrollIntoView({
      behavior: "smooth",
      block: "start",
    });
  };

  return (
    <main className="page">
      <section className="opening-section">
        <div className="opening-noise" />

        <div className="opening-content">
          <span className="opening-tag">Frame by frame experience</span>

          <h1 className="intro-title">
            O fim do vídeo
            <br />
            vira o começo da experiência.
          </h1>

          <p className="intro-text">
            Role a página e atravesse a cena frame por frame, como se o vídeo
            estivesse respondendo ao seu movimento.
          </p>

          <button onClick={scrollToExperience}>Começar</button>
        </div>
      </section>

      <section ref={containerRef} className="scroll-video-section">
        <canvas ref={canvasRef} className="scroll-canvas" />

        <div className="canvas-overlay" />

        <div ref={textRef} className="scroll-text">
          <span>Scroll cinematic</span>

          <h2>
            Cada rolagem revela
            <br />
            um novo instante.
          </h2>

          <p>
            São 240 frames conectados ao scroll, criando uma navegação com cara
            de filme.
          </p>
        </div>

        <div className="scroll-indicator">
          <span>role para avançar</span>
          <div />
        </div>
      </section>

      <section className="final-section">
        <div className="final-content">
          <span>Experiência concluída</span>

          <h2>
            Agora a página tem ritmo,
            <br />
            movimento e impacto.
          </h2>

          <p>
            Aqui você pode colocar o CTA final: contato, apresentação do projeto,
            assistir ao vídeo completo ou continuar navegando.
          </p>

          <button>Quero saber mais</button>
        </div>
      </section>
    </main>
  );
}

OBSERVAÇÃO IMPORTANTE

Os frames precisam estar exatamente aqui:

public/frames/inital/frame-00002.jpg
public/frames/inital/frame-00003.jpg
...
public/frames/inital/frame-00240.jpg

A pasta está escrita como "inital", igual você passou.
Se mudar para "initial", altere também no código:

return `/frames/initial/frame-${frameNumber}.jpg`;
